```
"""
Universal, target-agnostic HTML form parser for ModScan.

Goals:
- Robust extraction of forms across arbitrary HTML without target-specific logic.
- Capture action, method, enctype, and a normalized inputs map.
- Handle input, select (single/multiple), and textarea elements.
- Detect likely CSRF tokens using generic name-based patterns.
- Gracefully degrade if BeautifulSoup is unavailable using a regex/html.parser fallback.

API:
    parse_forms(html: str, base_url: Optional[str] = None) -> List[Dict]

Return structure example:
    {
      'action': 'https://example.com/login',
      'method': 'POST',
      'enctype': 'application/x-www-form-urlencoded',
      'inputs': {
         'username': {'type': 'text', 'value': ''},
         'password': {'type': 'password', 'value': ''},
         '__RequestVerificationToken': {'type':'hidden','value':'abc','is_csrf': True},
         'remember': {'type': 'checkbox', 'value': [], 'options': ['on']},
         'color': {'type': 'select', 'value': 'red', 'options': ['red','blue']}
      }
    }

Notes:
- For checkbox/radio groups, values are aggregated under a single name with:
  {'type': 'checkbox'|'radio', 'value': [checked_values], 'options': [all_values]}
- For <select multiple>, 'value' is a list of selected values.
- For <select>, 'options' holds a flat list of option values (text if no value attr).
- CSRF detection is generic by field name (csrf, xsrf, _token, authenticity_token, requestverificationtoken, etc.).
"""

from typing import List, Dict, Optional, Any
from urllib.parse import urljoin
import re

try:
    from bs4 import BeautifulSoup  # type: ignore
except Exception:  # bs4 may not be installed; fallback mode will be used
    BeautifulSoup = None  # type: ignore

MAX_HTML = 2_000_000  # Simple guard against pathological pages


_CSRF_NAME_PATTERN = re.compile(
    r"csrf|xsrf|_token|authenticity|requestverification|__requestverificationtoken",
    re.IGNORECASE,
)


def _is_csrf_field(name: Optional[str]) -> bool:
    if not name:
        return False
    # Normalize by removing common separators for broader matching
    norm = re.sub(r"[^a-z]", "", name.lower())
    return bool(_CSRF_NAME_PATTERN.search(norm))


def _finalize_action(action: Optional[str], base_url: Optional[str]) -> str:
    action = (action or "").strip()
    if not base_url:
        return action
    if not action:
        return base_url
    return urljoin(base_url, action)


def parse_forms(html: str, base_url: Optional[str] = None) -> List[Dict[str, Any]]:
    """Parse HTML and return a list of normalized form dicts.

    - Uses BeautifulSoup if available; otherwise, a robust regex fallback.
    - Never performs network operations or target-specific assumptions.
    """
    if not isinstance(html, str):
        html = str(html)
    if len(html) > MAX_HTML:
        html = html[:MAX_HTML]

    if BeautifulSoup is not None:
        try:
            return _parse_with_bs4(html, base_url)
        except Exception:
            # Fall back if BS4 chokes on malformed HTML
            pass
    return _parse_with_fallback(html, base_url)


def _parse_with_bs4(html: str, base_url: Optional[str]) -> List[Dict[str, Any]]:
    soup = BeautifulSoup(html, "html.parser")  # type: ignore
    results: List[Dict[str, Any]] = []
    for form in soup.find_all("form"):
        action = _finalize_action(form.get("action"), base_url)
        method = (form.get("method") or "GET").upper()
        enctype = (form.get("enctype") or "application/x-www-form-urlencoded").strip()

        inputs: Dict[str, Any] = {}

        # Iterate over all input/select/textarea descendants
        for el in form.find_all(["input", "select", "textarea"]):
            tag = el.name.lower()
            name = el.get("name")
            if not name:
                continue  # skip unnamed controls

            if tag == "input":
                itype = (el.get("type") or "text").lower()
                raw_val = el.get("value")
                is_csrf = _is_csrf_field(name)

                if itype in ("checkbox", "radio"):
                    val = raw_val if raw_val is not None else "on"
                    item = inputs.get(name)
                    if not item:
                        item = {"type": itype, "value": [], "options": [], "is_csrf": is_csrf}
                        inputs[name] = item
                    # Track all options
                    if val not in item["options"]:
                        item["options"].append(val)
                    # Track checked values
                    if el.has_attr("checked"):
                        if val not in item["value"]:
                            item["value"].append(val)
                    # Provide a summarizing boolean checked flag for simple UIs
                    item["checked"] = bool(item["value"])  # True if any checked
                else:
                    # Single value input
                    if name not in inputs:
                        inputs[name] = {
                            "type": itype,
                            "value": raw_val if raw_val is not None else "",
                            "is_csrf": is_csrf,
                        }
                    # First one wins for duplicate text inputs; later duplicates ignored

            elif tag == "textarea":
                is_csrf = _is_csrf_field(name)
                value = el.text if el.text is not None else ""
                if name not in inputs:
                    inputs[name] = {"type": "textarea", "value": value, "is_csrf": is_csrf}

            elif tag == "select":
                is_csrf = _is_csrf_field(name)
                multiple = el.has_attr("multiple")
                options: List[str] = []
                selected: List[str] = []
                for opt in el.find_all("option"):
                    opt_val = opt.get("value")
                    if opt_val is None:
                        opt_val = (opt.text or "").strip()
                    options.append(opt_val)
                    if opt.has_attr("selected"):
                        selected.append(opt_val)
                if multiple:
                    value: Any = selected
                else:
                    value = selected[0] if selected else (options[0] if options else "")
                if name not in inputs:
                    item = {
                        "type": "select",
                        "value": value,
                        "options": options,
                        "multiple": multiple,
                        "is_csrf": is_csrf,
                    }
                    # Also expose selected options list for clarity
                    item["selected_options"] = list(selected)
                    inputs[name] = item

        results.append({
            "action": action,
            "method": method,
            "enctype": enctype,
            "inputs": inputs,
        })

    return results


# --- Fallback parser (regex-based, best-effort) ---

_ATTR_RE = re.compile(r"(\w[\w:\-]*)\s*=\s*(?:'([^']*)'|\"([^\"]*)\"|([^\s'\">]+))")


def _parse_attrs(attr_text: str) -> Dict[str, str]:
    attrs: Dict[str, str] = {}
    for m in _ATTR_RE.finditer(attr_text or ""):
        key = m.group(1).lower()
        val = m.group(2) or m.group(3) or m.group(4) or ""
        attrs[key] = val
    return attrs


def _parse_with_fallback(html: str, base_url: Optional[str]) -> List[Dict[str, Any]]:
    results: List[Dict[str, Any]] = []
    # Find forms and their contents
    for fm in re.finditer(r"(?is)<form\b([^>]*)>(.*?)</form>", html):
        form_attr_txt = fm.group(1) or ""
        form_inner = fm.group(2) or ""
        fattrs = _parse_attrs(form_attr_txt)

        action = _finalize_action(fattrs.get("action"), base_url)
        method = (fattrs.get("method") or "GET").upper()
        enctype = (fattrs.get("enctype") or "application/x-www-form-urlencoded").strip()

        inputs: Dict[str, Any] = {}

        # Inputs
        for im in re.finditer(r"(?is)<input\b([^>]*)>", form_inner):
            iattrs = _parse_attrs(im.group(1) or "")
            name = iattrs.get("name")
            if not name:
                continue
            itype = (iattrs.get("type") or "text").lower()
            raw_val = iattrs.get("value")
            is_csrf = _is_csrf_field(name)
            checked = re.search(r"(?is)\bchecked\b", im.group(1) or "") is not None

            if itype in ("checkbox", "radio"):
                val = raw_val if raw_val is not None else "on"
                item = inputs.get(name)
                if not item:
                    item = {"type": itype, "value": [], "options": [], "is_csrf": is_csrf}
                    inputs[name] = item
                if val not in item["options"]:
                    item["options"].append(val)
                if checked and val not in item["value"]:
                    item["value"].append(val)
                item["checked"] = bool(item["value"])  # summarizing flag
            else:
                if name not in inputs:
                    inputs[name] = {"type": itype, "value": raw_val or "", "is_csrf": is_csrf}

        # Textareas
        for tm in re.finditer(r"(?is)<textarea\b([^>]*)>(.*?)</textarea>", form_inner):
            tattrs = _parse_attrs(tm.group(1) or "")
            name = tattrs.get("name")
            if not name:
                continue
            value = (tm.group(2) or "")
            is_csrf = _is_csrf_field(name)
            if name not in inputs:
                inputs[name] = {"type": "textarea", "value": value, "is_csrf": is_csrf}

        # Selects
        for sm in re.finditer(r"(?is)<select\b([^>]*)>(.*?)</select>", form_inner):
            sattrs = _parse_attrs(sm.group(1) or "")
            name = sattrs.get("name")
            if not name:
                continue
            multiple = "multiple" in (sm.group(1) or "").lower()
            options: List[str] = []
            selected: List[str] = []
            for om in re.finditer(r"(?is)<option\b([^>]*)>(.*?)</option>", sm.group(2) or ""):
                oattrs = _parse_attrs(om.group(1) or "")
                val = oattrs.get("value")
                if val is None:
                    val = (om.group(2) or "").strip()
                options.append(val)
                if re.search(r"(?is)\bselected\b", om.group(1) or ""):
                    selected.append(val)
            value: Any
            if multiple:
                value = selected
            else:
                value = selected[0] if selected else (options[0] if options else "")
            is_csrf = _is_csrf_field(name)
            if name not in inputs:
                item = {
                    "type": "select",
                    "value": value,
                    "options": options,
                    "multiple": multiple,
                    "is_csrf": is_csrf,
                }
                item["selected_options"] = list(selected)
                inputs[name] = item

        results.append({
            "action": action,
            "method": method,
            "enctype": enctype,
            "inputs": inputs,
        })

    return results
```

```
def _run(coro):
    loop = asyncio.new_event_loop()
    try:
        return loop.run_until_complete(coro)
    finally:
        loop.close()

      <input type="submit" name="btn" value="Go">
    </form>
    """
    forms = _run(scanner._parse_forms_with_values(html))
    assert len(forms) == 1
    inputs = forms[0]["inputs"]

    # Selection per SQLi path (same preserved fields)
    testable = []
    protected = {}
    for name, data in inputs.items():
        t = (data.get("type") or "text").lower()
        is_csrf = bool(data.get("is_csrf"))
        if t in ["hidden", "submit", "button"] or is_csrf:
            protected[name] = data.get("value")
        else:
            testable.append(name)

    assert "q" in testable
    assert "_token" in protected
    assert "btn" in protected
```
